### 问题提出

假设固定几台代理服务器，对音视频流进行透明代理转发，希望转发性能越高越好。

假设一台代理服务器负责5万路码率为1Mbps的音视频流的转发，udp包大小为800B，那么这台代理服务器的转发性能需要达到：

```
50000 x 1000000 / 8 / 800 = 800万pps
```

### 方案一：利用iptables设置转发规则，内核态转发

三台机器 A，B，C，2c4g，30万pps+100Mbps外网带宽，A运行udp客户端，B负责代理，C运行udp服务端。

#### step1：每台机器设置套接字缓冲区大小，避免流量抖动的时候丢包：

修改/etc/sysctl.conf

```
net.core.rmem_default=1000000
net.core.rmem_max=1000000
net.core.wmem_default=1000000
net.core.wmem_max=1000000
```

然后执行：

```
sudo sysctl -p
```

#### step2：机器B设置转发模式，并且设置iptables：

```
net.ipv4.ip_forward=1
net.ipv6.conf.all.forwarding=1
```

然后执行：

```shell
sudo sysctl -p
```



```shell
#!/usr/bin/sh
A_IP=49.232.243.192
B_IP=10.0.1.13
C_IP=43.143.234.162

sudo iptables -t nat -D PREROUTING -s  ${A_IP} -p udp --dport 7890 -j DNAT --to-destination ${C_IP}
sudo iptables -t nat -D POSTROUTING -d ${C_IP} -p udp --dport 7890 -j SNAT --to-source  ${B_IP}

sudo iptables -t nat -A PREROUTING -s  ${A_IP} -p udp --dport 7890 -j DNAT --to-destination ${C_IP}
sudo iptables -t nat -A POSTROUTING -d ${C_IP} -p udp --dport 7890 -j SNAT --to-source  ${B_IP}


sudo iptables -t nat -D POSTROUTING -s ${C_IP} -p udp --sport 7890 -j SNAT --to-source ${B_IP}
sudo iptables -t nat -A POSTROUTING -s ${C_IP} -p udp --sport 7890 -j SNAT --to-source ${B_IP}

```

服务端：

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>

#define SERVER_PORT 7890
#define BUFFER_SIZE 1024

int main() {
    int sockfd;
    struct sockaddr_in server_addr, client_addr;
    char buffer[BUFFER_SIZE];
    socklen_t addr_len;


    // 创建UDP套接字
    sockfd = socket(AF_INET, SOCK_DGRAM, 0);
    if (sockfd < 0) {
        perror("socket creation failed");
        exit(EXIT_FAILURE);
    }

    // 设置服务器地址
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(SERVER_PORT);

    // 将套接字绑定到服务器地址
    if (bind(sockfd, (struct sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
        perror("bind failed");
        exit(EXIT_FAILURE);
    }

    printf("UDP Echo服务器正在运行，监听端口：%d\n", SERVER_PORT);

    for (int i = 0; i < 100; ++i)
    {
        if (fork() == 0)
        {
            break;
        }
    }

    while (1) {
        memset(buffer, 0, sizeof(buffer));
        addr_len = sizeof(client_addr);

        // 接收来自客户端的消息
        ssize_t recv_len = recvfrom(sockfd, buffer, BUFFER_SIZE, 0, (struct sockaddr*)&client_addr, &addr_len);
        if (recv_len < 0) {
            perror("recvfrom failed");
            exit(EXIT_FAILURE);
        }

#if 1
        // 将接收到的消息发送回客户端
        if (sendto(sockfd, buffer, strlen(buffer), 0, (struct sockaddr*)&client_addr, addr_len) < 0) {
            perror("sendto failed");
            exit(EXIT_FAILURE);
        }
#endif

    }

    // 关闭套接字
    close(sockfd);

    return 0;
}

```
